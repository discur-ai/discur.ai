```astro
---
interface Props {
  id?: string;
  class?: string;
  fullWidth?: boolean;
  dataAnimate?: string;
}

const { id, class: className, fullWidth = false, dataAnimate } = Astro.props;
---

<section
  id={id}
  class:list={[
    'py-section px-4 sm:px-6 lg:px-8',
    !fullWidth && 'mx-auto max-w-4xl',
    className,
  ]}
  data-animate={dataAnimate}
>
  <slot />
</section>
```
